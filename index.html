Comportamento

In other words, don't check whether it IS-a duck: check whether it QUACKS-like-a duck, WALKS-like-a duck, etc, etc, [...]

- Alex Martelli

class Duck(object):
    def walk(self):
        pass

    def quack(self):
        pass


def quack(duck):
    if isinstance(duck, Duck):
        duck.quack()

class Duck(object):
    def walk(self):
        pass

    def quack(self):
        pass


def quack(duck):
    if hasattr(duck, "quack"):
        duck.quack()
